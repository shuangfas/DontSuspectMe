DSM_increase_benefits_level = {
	hidden_effect = {
		add_to_variable = {
			var = benefits_level
			value = 0.1
		}
		if = {
			limit = {
				NOT = {
					has_dynamic_modifier = DSM_export_adjustment
				}
			}
			add_dynamic_modifier = {
				modifier = DSM_export_adjustment
			}
		}
	}
	set_temp_variable = {
		var = export_show_temp
		value = benefits_level
	}
	add_to_temp_variable = {
		var = export_show_temp
		value = 0.1
	}
	set_temp_variable = {
		var = temp_num
		value = -0.1
	}
	DSM_increase_suspicion_weekly_factor_modifier = yes
	custom_effect_tooltip = DSM_export_adjustment_increase_desc
}

DSM_decrease_benefits_level = {
	hidden_effect = {
		if = {
			limit = {
				check_variable = {
					var = benefits_level
					value = 0.1
					compare = less_than_or_equals
				}
			}
			remove_dynamic_modifier = {
				modifier = DSM_export_adjustment
			}
		}
		add_to_variable = {
			var = benefits_level
			value = -0.1
		}
	}
	set_temp_variable = {
		var = export_show_temp
		value = benefits_level
	}
	add_to_temp_variable = {
		var = export_show_temp
		value = -0.1
	}
	set_temp_variable = {
		var = temp_num
		value = 0.1
	}
	DSM_increase_suspicion_weekly_factor_modifier = yes
	custom_effect_tooltip = DSM_export_adjustment_decrease_desc
}

DSM_all_country_declare_war = {

			}
			declare_war_on = {
				target = ATW
				type = annex_everything
			}
		}
	}
}
DSM_calculate_civilian_intel_factor = {
	set_variable = {
		var = ATW.DSM_civilian_intel_sum
		value = 0
	}
	set_variable = {
		var = ATW.DSM_country_num
		value = 0
	}
	every_country = {
		limit = {
			NOT = {
				tag = ATW
			}
			is_major = yes
			has_intelligence_agency = yes
		}
		set_temp_variable = {
			var = temp_intel
			value = THIS.civilian_intel@ATW
		}
		add_to_variable = {
			var = ATW.DSM_country_num
			value = 1
		}
		clamp_temp_variable = {
			var = temp_intel
			min = 0
			max = 100
		}
		ATW = {
			add_to_variable = {
				var = ATW.DSM_civilian_intel_sum
				value = temp_intel
			}
		}
	}
}
DSM_increase_suspicion_weekly_factor_modifier = {
	add_to_variable = {

	}
	custom_effect_tooltip = DSM_suspicion_weekly_factor_modifier

}
DSM_calculate_suspicion_weekly_factor = {
	add_to_temp_variable = {
		var = factor_temp
		value = DSM_civilian_intel_sum
	}
	multiply_temp_variable = {
		var = factor_temp
		value = 0.008
	}
	divide_temp_variable = {
		var = factor_temp
		value = DSM_country_num
	}
	add_to_temp_variable = {
		var = factor_temp
		value = suspicion_weekly_factor_modifier
	}
	add_to_temp_variable = {
		var = factor_temp
		value = 1
	}
	set_variable = {
		var = suspicion_weekly_factor
		value = factor_temp
	}
	set_variable = {
		var = ATW_politics_suspicion_weekly_dynamic_var
		value = 0.005
	}
	multiply_variable = {
		var = ATW_politics_suspicion_weekly_dynamic_var
		value = suspicion_weekly_factor
	}
}
