DSM_calcucate_discount = {
	set_variable = {
		var = DSM_dynamic_cost_ratio
		value = 0
	}
	add_to_variable = {
		var = DSM_dynamic_cost_ratio
		value = ATW_suspicion_var
	}
	subtract_from_variable = {
		var = DSM_dynamic_cost_ratio
		value = 0.25
	}
	multiply_variable = {
		var = DSM_dynamic_cost_ratio
		value = 2
	}
	add_to_variable = {
		var = DSM_dynamic_cost_ratio
		value = 1
	}
	clamp_variable = {
		var = DSM_dynamic_cost_ratio
		min = 0.4
		max = 5
	}
}

#Passed-in original cost to DSM_temp_cost temp_variable
DSM_apply_cost_pp = {
	log = "[?DSM_temp_cost]"
	multiply_temp_variable = {
		var = DSM_temp_cost
		value = DSM_dynamic_cost_ratio
	}
	multiply_temp_variable = {
		var = DSM_temp_cost
		value = -1
	}
	hidden_effect = {
		add_political_power = DSM_temp_cost
	}
}

DSM_calculate_disarmament_require = {
	set_variable = {
		var = DSM_disarmament_require
		value = deployed_army_manpower_k
	}
	multiply_variable = {
		var = DSM_disarmament_require
		value = 700		#0.6*k
	}
	log = "[?DSM_disarmament_require]"
}

DSM_activate_disarmament = {
