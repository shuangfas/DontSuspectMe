ATW_add_resistance = {
	custom_effect_tooltip = ATW_add_resistance_tooltip
	add_to_variable = {
		var = ATW_resistance_var
		value = ATW_resistance_var_temp
	}
	clamp_variable = {
		var = ATW_resistance_var
		min = 0
		max = 1
	}
	hidden_effect = {
		remove_ideas = ATW_lower_resistance_idea
		remove_ideas = ATW_low_resistance_idea
		remove_ideas = ATW_medium_resistance_idea
		remove_ideas = ATW_high_resistance_idea
		remove_ideas = ATW_higher_resistance_idea
		if = {
			limit = {
				check_variable = {
					var = ATW_resistance_var
					value = 0.2
					compare = less_than
				}
			}
			add_ideas = ATW_lower_resistance_idea
		}
		else_if = {
			limit = {
				check_variable = {
					var = ATW_resistance_var
					value = 0.2
					compare = greater_than_or_equals
				}
				check_variable = {
					var = ATW_resistance_var
					value = 0.4
					compare = less_than
				}
			}
			add_ideas = ATW_low_resistance_idea
		}
		else_if = {
			limit = {
				check_variable = {
					var = ATW_resistance_var
					value = 0.4
					compare = greater_than_or_equals
				}
				check_variable = {
					var = ATW_resistance_var
					value = 0.6
					compare = less_than
				}
			}
			add_ideas = ATW_medium_resistance_idea
		}
		else_if = {
			limit = {
				check_variable = {
					var = ATW_resistance_var
					value = 0.6
					compare = greater_than_or_equals
				}
				check_variable = {
					var = ATW_resistance_var
					value = 0.8
					compare = less_than
				}
			}
			add_ideas = ATW_high_resistance_idea
		}
		else_if = {
			limit = {
				check_variable = {
					var = ATW_resistance_var
					value = 0.8
					compare = greater_than_or_equals
				}
			}
			add_ideas = ATW_higher_resistance_idea
		}
	}
}

	custom_effect_tooltip = ATW_add_suspicion_tooltip
	add_to_variable = {
		var = ATW_suspicion_var
		value = ATW_suspicion_var_temp
	}
	clamp_variable = {
		var = ATW_suspicion_var
		min = 0
		max = 1
	}
	if = {
		limit = {
			OR = {
				is_in_faction_with = NSM
				is_in_faction_with = PTF
			}
		}
		if = {
			limit = {
				check_variable = {
					var = ATW_suspicion_var
					value = 0.65
					compare = greater_than_or_equals
				}
			}
			activate_mission = ATW_the_boot_mission
		}
		if = {
			limit = {
				check_variable = {
					var = ATW_suspicion_var
					value = 0.5
					compare = greater_than_or_equals
				}
			}
			country_event = atom.69
		}
	}
	else_if = {
		limit = {
			has_country_flag = ATW_on_the_world_stage_flag
		}
		if = {
			limit = {
				check_variable = {
					var = ATW_suspicion_var
					value = 0.6
					compare = greater_than_or_equals
				}
			}
			if = {
				limit = {
					NOT = {
						has_active_mission = ATW_international_intervention_mission
					}
				}
				activate_mission = ATW_international_intervention_mission
			}
			if = {
				limit = {
					NOT = {
						has_active_mission = DSM_disarmament_misson
					}
				}
				country_event = {
					id = dsm.4
					days = 1
				}
			}
		}
		if = {
			limit = {
				check_variable = {
					var = ATW_suspicion_var
					value = 0.8
					compare = greater_than_or_equals
				}
			}
			activate_mission = ATW_popular_revolt_mission
		}
		if = {
			limit = {
				check_variable = {
					var = ATW_suspicion_var
					value = 0.99
					compare = greater_than_or_equals
				}
			}
			activate_mission = ATW_popular_revolt_mission
		}
	}
