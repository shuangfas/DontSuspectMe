	name = DSM_dynamic_cost_50_pp_and_50_cp
	text = {
		trigger = {
			set_temp_variable = {
				var = DSM_temp_cost
				value = 50
			}
			multiply_temp_variable = {
				var = DSM_temp_cost
				value = DSM_dynamic_cost_ratio
			}
			always = yes
		}
		localization_key = DSM_dynamic_cost_pp_and_cp
